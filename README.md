# typst是图文样式设计器，基于tinymce开发，支持多种图文样式，可以快速设计出美观的文章。

# 进入项目目录
cd typst

# 安装依赖
npm install

# 建议不要直接使用 cnpm 安装以来，会有各种诡异的 bug。可以通过如下操作解决 npm 下载速度慢的问题
npm install --registry=https://registry.npm.taobao.org

# 启动服务
npm run dev
